class Solution {
    public int lengthOfLongestSubstring(String s) {
        Set<Character> set = new HashSet<Character>();
        int longest = 0;
        int i=0,j=0,n=s.length();
        while(i<n && j<n)
        {
            if(!set.contains(s.charAt(j)))
                set.add(s.charAt(j++));
            else {
                set.remove(s.charAt(i++));
            }
            longest = Math.max(set.size(), longest);
        }
        return longest;
    }
}
